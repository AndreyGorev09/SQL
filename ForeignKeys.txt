# СОЗДАДИМ БАЗУ ДАННЫХ CONTACTS_DB И ТАБЛИЦУ В НЕЙ CONTACS:
CREATE DATABASE contacts_db;
USE contacts_db;
CREATE TABLE contacts (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(25) NOT NULL);
DESC contacts;
